# 3. 音频和视频

# 音频标签audio

# audio标签是HTML5的新标签（双标签），用于定义声音，比如音乐或其他音频流。支持的3种文件格式：MP3、Wav、Ogg。

<audio src="./music/1.mp3" controls="controls">
	<span>你的浏览器不支持audio标签。</span>
</audio>
说明

  • controls 属性供添加播放、暂停和音量控件。

  • audio标签中间插入的内容是供不支持 audio元素的浏览器显示的

<audio controls="controls">
    <source src="star.ogg" type="audio/ogg">
    <source src="star.mp3" type="audio/mpeg">
    <span>你的浏览器不支持audio标签。</span>
</audio>
说明

source子标签（单标签）来链接不同的音频文件，浏览器将使用第一个可识别的格式。

# audio属性:

属性 值 描述
autoplay autoplay 如果出现该属性，则音频在就绪后马上播放。（此属性已被主流浏览器禁止）
controls controls 如果出现该属性，则向用户显示控件，比如播放按钮。
loop loop 如果出现该属性，则每当音频结束时重新开始播放
muted muted 规定视频输出应该被静音。
src url 要播放的音频的URL地址。

# 4. Canvas画布

传统的网页，总是使用GIF或者JPEG来显示图像，这种图形是需要事先画好的“静态”的图像。而Canvas则是用Javascript的一种绘图手段。可以用它来画图、合成图象、或做动画。

Canvas元素本身没有绘制能力，它仅仅是图形的容器。要完成实际的绘图功能要使用Javascript语言。

# 绘制线段

<canvas id="cav" width="400" height="200">你的浏览器不支持此功能</canvas>
<script>
    //获取容器
    let cav = document.getElementById('cav');
    cav.style.backgroundColor = 'lightblue';
    //获得2d绘图环境的绘图对象
    let ctx = cav.getContext('2d');
    //开始绘制
    ctx.beginPath();
    // 设置路径起点坐标为
    ctx.moveTo(50, 50);
    // 绘制直线中间的点，可以有多个
    ctx.lineTo(100, 100);
    ctx.lineTo(150, 100);
    // 设置线宽
    ctx.lineWidth = 4;
    // 设置线的颜色
    ctx.strokeStyle = '#CC0000';
    // stroke方法用于绘制线段（只绘制线段，不填充）
    ctx.stroke();
    //结束绘制
    ctx.closePath();
</script>

# 绘制圆形

<canvas id="cav" width="500" height="200">你的浏览器不支持此功能</canvas>
<script>
    let cav = document.getElementById('cav');
    cav.style.backgroundColor = 'lightblue';
    let ctx = cav.getContext('2d');
    ctx.beginPath();
    //填充颜色
    ctx.fillStyle = 'blue';
    //设置圆形参数（圆心x、圆心y、半径、开始角度、结束角度(PI为180度)、是否逆时针方向绘制）
    ctx.arc(100, 80, 60, 0, Math.PI * 2, true);
    //绘制圆形
    ctx.fill();
    ctx.closePath();
</script>

# 绘制时钟

<canvas width="1200" height="1200" id="canvas"></canvas>
<script>
  let clock = document.getElementById('canvas').getContext('2d')
  let minuteDeg = - Math.PI / 2
  let hoursDeg = - Math.PI / 2
  let secondDeg = - Math.PI / 2

  setInterval(run, 1000)
  function run() {
    // 重新绘制
    clock.clearRect(0, 0, 1200, 1200)
    clock.beginPath()
    clock.arc(320, 320, 300, 0, Math.PI * 2, true)
    clock.stroke()
    clock.closePath()
    
    let a = 0
    let b = Math.PI / 30
    for (let i = 0; i < 60; i++) {
      clock.beginPath()
      clock.arc(320, 320, 300, a, a, true)
      clock.arc(320, 320, i % 5 == 0 ? 280 : 290, a, a, true)
      clock.stroke()
      clock.closePath()
      a += b
    }

    let hoursX = 320 + 120 * Math.cos(hoursDeg)
    let hoursY = 320 + 120 * Math.sin(hoursDeg)

    let minuteX = 320 + 180 * Math.cos(minuteDeg)
    let minuteY = 320 + 180 * Math.sin(minuteDeg)

    let secondX = 320 + 240 * Math.cos(secondDeg)
    let secondY = 320 + 240 * Math.sin(secondDeg)

    // 时针
    clock.beginPath()
    clock.moveTo(320, 320)
    clock.lineTo(hoursX, hoursY)
    clock.lineWidth = 6
    clock.stroke()
    clock.closePath()

    // 分针
    clock.beginPath()
    clock.moveTo(320, 320)
    clock.lineTo(minuteX, minuteY)
    clock.lineWidth = 4
    clock.stroke()
    clock.closePath()

    // 秒针
    clock.beginPath()
    clock.moveTo(320, 320)
    clock.lineTo(secondX, secondY)
    clock.lineWidth = 1
    clock.stroke()
    clock.closePath()

    secondDeg += Math.PI * 2 / 60
    minuteDeg += Math.PI * 2 / 60 / 60
    hoursDeg +=  Math.PI * 2 / 12 / 60 / 60
  }
</script>

# 5. SVG绘图

SVG：可伸缩矢量图（Scalable Vector Graphics）：SVG图形是矢量图，所以在放大缩小的情况下，不会损失图形质量，这一点与依靠像素等位图是不同。对比：Canvas依赖分辨率，而SVG不依赖分辨率

# 6. 地理定位

地理位置（Geolocation）是 HTML5 的重要特性之一，提供了确定用户位置的功能，借助这个特性能够开发基于位置信息的应用。这个功能在移动端尤其实用。

说明
  • 注意：地理定位需要基于服务器运行

# 地理定位实例

说明

  • 使用Edge浏览器打开，Chrome浏览器可能会拒绝访问

  • 使用navigator.geolocation来获取HTML5的地理定位对象

  • 使用navigator.geolocation对象的getCurrentPosition() 方法获得用户的位置，有三个参数：

    • 获得地理位置成功时的回调函数。
    • 获得地理位置失败时的回调函数
    • 配置信息，使用JSON格式。
<script>
    // 使用navigator.geolocation来获取HTML5的地理定位对象
    if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(function (position) {
            //获取纬度
            let latitude = position.coords.latitude;
            //获取经度
            let longitude = position.coords.longitude;
            alert('纬度：' + latitude + '，经度：' + longitude);
        }, function (error) {
            switch (error.code) {
                case error.TIMEOUT:
                    alert('获得地理位置超时!');
                    break;
                case error.POSITION_UNAVAILABLE:
                    alert('不能获取位置！');
                    break;
                case error.PERMISSION_DENIED:
                    alert('请开启地理定位功能！');
                    break;
                case error.UNKNOWN_ERROR:
                    alert('未知错误！');
                    break;
            }
        }, {
            // 指定获取地理位置的超时时间，默认不限时，单位为毫秒
            timeout: 5000,
        });
    } else {
        alert('你的浏览器不支持地理定位！');
    }
</script>

# 7. 拖放API

实现鼠标拖放的三个事件：

  • ondragstart：开始拖动事件。

  • ondragover：被拖动对象移动到目标对象上事件。

  • ondrop：被拖动对象放下事件。

示例代码 
<style>
    .box {
      width: 400px;
      height: 200px;
      background-color: skyblue;
    }
    .active {
      width: 100px;
      height: 60px;
      background-color: red;
      line-height: 60px;
      text-align: center;
      color: #fff;
      cursor: default;
      /*  设置不能选择文字*/
      user-select: none;
    }
 </style>
 <body style="height: 100vh;">
  <div class="box"></div>
  <div class="active" draggable="true">
    被拖拽对象
  </div>
  <script>
    let box  = document.querySelector('.box')
    let active  = document.querySelector('.active')

    //被拖拽的元素必须有 draggable="true" 属性
    active.ondragstart = function () {
      console.log("我被拖拽了")
    }

    //ondragover事件有浏览器默认行为，所以要阻止此默认行为。
    document.body.ondragover = function (event) {
      event.preventDefault()
    }
    document.body.ondrop = function (event) {
      event.target.appendChild(active)
    }
  </script>
</body>

# 8. WebWorker

WebWorker 是一个javascript多线程解决方案，它真正实现了多线程。

具体来说：它允许在 Web 程序中并发执行多个 JavaScript 脚本，每个脚本执行流都称为一个线程，彼此间互相独立，并且有浏览器中的 JavaScript 引擎负责管理。

# 创建Test1.js 文件

for(let i=0;i<100;i++){
    console.log('线程1：'+i);
}

# 创建Test2.js 文件

for(let i=0;i<100;i++){
    console.log('线程2：'+i);
}

# 创建index.html 文件

<script>
    let worker1 = new Worker("./js/Test1.js");
    let worker2 = new Worker("./js/Test2.js");
</script>

  • new Worker() 就相当于创建一个新的线程。

  • 仔细观察运行结果，就会发现两个线程是同时运行的。所以运行结果不确定。

# 获取线程执行结果

Web Worker提供了一个事件和两个方法，用于操作线程：

  • onmessage事件：当回传消息时触发此事件。

  • postMessage()方法：回传消息。

  • terminate()方法：终止WebWorker。

# 修改Test1.js 文件，添加回传消息方法：

for(let i=0;i<100;i++){
    console.log('线程1：'+i);
}
postMessage('线程1运行完毕');

# 修改Test2.js 文件，添加回传消息方法：

for(let i=0;i<100;i++){
    console.log('线程2：'+i);
}
postMessage('线程2运行完毕');

# 修改index.html 文件，添加onmessage事件：

<script>
    let worker1 = new Worker("./js/Test1.js");
    let worker2 = new Worker("./js/Test2.js");
    //线程1的onmessage事件
    worker1.onmessage = function (event) {
        console.log(event.data);
    }
    //线程2的onmessage事件
    worker2.onmessage = function (event) {
        console.log(event.data);
    }
</script>

​ 从运行结果中可以看到：每个线程的回传消息时间也是不确定的。

# WebStorage

WebStorage（Web存储）提供了两种在客户端存储数据的新方法：

  • localStorage：用于持久化的本地存储，除非主动删除数据，否则数据是永远不会过期。

  • sessionStorage：用于存储一个会话（session）范围中的数据（即 从浏览器打开页面开始，到浏览器页面关闭为止），这些数据只有在同一个会话中的页面才能访问，当会话结束后数据也随之销毁。

# SessionStorage：

SessionStorage 中有三个主要方法，能够进行数据的存储：

  1. setItem()： 该方法接受一个键和值作为参数，将会把键值对添加到存储中，如果键名存在，则更新其对应的值。
  2. getItem()：该方法接受一个键名作为参数，返回键名对应的值。
  3. removeItem()：该方法接受一个键名作为参数，并把该键名从存储中删除。
代码示例： 
//向sessionStorage存入数据
sessionStorage.setItem('user','狗蛋');
//从sessionStorage中取出数据
let str = sessionStorage.getItem('user');
console.log(str);       //狗蛋

代码运行后，在浏览器控制台的 👉 Application中 👉 Storage 👉 Session Storage就可以看到存储结果

⚠️

注意：当关闭浏览器后，再次打开浏览器中查看，就会发现SessionStorage中存储的数据已经没有了。那是因为，当浏览器关闭后，一个session会话范围就结束了，那么SessionStorage中的数据自然也就没有了。

# LocalStorage

LocalStorage：永久存储。只要不主动从LocalStorage中将数据移除，那么LocalStorage种的数据就始终存在。

代码示例： 
//向localStorage存入数据
localStorage.setItem('user','狗蛋');
//从localStorage中取出数据
let str = localStorage.getItem('user');
console.log(str);  //狗蛋

代码运行后，在浏览器控制台的 👉 Application中 👉 Storage 👉 Local Storage就可以看到存储结果

⚠️

注意：当关闭浏览器后，再次打开浏览器中查看，就会发现LocalStorage中存储的数据仍然还在。

# 存储对象

SessionStorage 和LocalStorage中只能存储字符串数据。但我们可能需要存储对象，此时可以将对象转换为字符串进行存储，待取出时，再将字符串转换为对象。

代码示例： 
let user = {
	userId:1,
	userName:'张三',
	userSex:'男'
};   
//将对象转换成字符串，进行存储
sessionStorage.setItem('user',JSON.stringify(user));
//将取出的字符串，转换成对象	
let obj = JSON.parse(sessionStorage.getItem('user'));
console.log(obj);
