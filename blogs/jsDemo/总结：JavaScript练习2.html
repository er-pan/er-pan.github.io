function deepCopy(source, target) { //source--数据源，从哪里拷贝，target--目标--拷贝到哪里去
    //循环遍历
    for(var key in source) { //in 遍历所有，不管是私有还是公有
        //判断只要私有的属性
        if(source.hasOwnProperty(key)) { //只拷贝私有属性，不再拷贝原型上的属性
            //判断是否引用类型
            if(typeof source[key] == "object") { //此时表示的是引用类型
            	//console.log(source[key]) //测试

                //target[key] = {} //拿一个对象来接受
                //三元预算符
                target[key] = Array.isArray(source[key]) ? [] : {}
                //递归
                deepCopy(source[key], target[key])
            }else {
                //则是拷贝值类型
                target[key] = source[key]
            }
        }
    }
}

/* 
单线程和多线程
JS是单线程:同一个时间只能执行一个任务
操作系统是多线程 c++

同步编程和异步编程
同步编程(sync)--JS里面的任务会按照顺序依次执行,当前的任务没有完成,后面的任务是不会执行的（ 同一个时间只能做一个事情）我们的同步任务会放在主任务队列里面。

异步编程(async)--JS当遇到异步任务的时候,不会停止任务，会吸纳把异步编程放到等待任务队列里面。程序会继续往下执行。先把主任务队列里面的程序执行完毕之后，并且到了等待时间，在继续去执行异步任务（"机器人去等待任务队列里面去找异步任务，找到异步任务之后要把异步任务放到主任务队列里面去执行"）

总结：
 同步任务都是放到主任务队列里面
 异步任务都是放到等待任务队列的

 JS里面哪些是同步哪些是异步呢？
 JS的异步编程
 第一、定时器
 第二、点击事件、普通事件都是异步
 第三、Ajax中一般也使用异步编程
 
 其余的都是同步的
*/

/*

举例："我的电话：10086，你的电话：10010，他的电话：10000" 我们会写一个正则表达式来匹配这个字符串或者说把这个字符串中的所有的数字都找到。

正则表达式的组成：是由元字符或者限定符组成的一个式子

******** 
元字符（具有特殊含义的字符）---建议可以参考百度、谷歌或者MDN
. 除了换行以外的所有的字符（任意一个字符） 举例: "abcedfg12345" 可以匹配的
[] 表示的是范围, [0-9] 表示的是0-9之间的任意的一个数字,可以是0，也可以是9 "789" [0-9] 
[1-7] 表示的是1到7之间的任意的一个数字（考题：可以不可以这样写正则去匹配100到200之间的数字的范围 [100-200]--这个写法是错误的，工作里面的写法--了解 ^(1\d{2}|200)$ --解释：分两种情况, 开头是1,后面两个数字随便,开头是2,后面只能写00）---等明天回来再看秒懂

[A-Z] 表示的是: 所有大写字母中的任意的一个字母
[a-zA-Z] 表示的是: 所有的字母中的任意的一个
[0-9a-zA-Z] 表示的是：所有的数字或者字母中的任意的一个

| 代表或者的意思
[0-9]|[a-z] 表示的是：要么是一个数字，要么是一个小写的字符
() 分组，提升优先级 举例 [0-9]|([a-z])|[A-Z] 此时会先算小括号里面的 
([0-9])([1-5])([a-z]) 三组，从左面开始计算

量词元字符 (也有人会叫做限定符)
* 代表前面字符出现了0次或者多次 举例 [a-z][0-9]* 表示的是小写字母中的任意的一个,后面要么没有数字，要么多个数字
+ 代表前面的字符出现了1次或者多次 举例 [a-z][9]+ 表示小写字母一个后面跟一个9或者多个9
? 代表前面的字符出现了0次或者1次 举例 [6][a-z]? 表示前面是6后面有字母或者没有都可以

{1,} 代表的是前面的表达式出现了1次或者多次 和 + 是一样的

{n} 代表的前面的字符连续出现了n次， 例如  {5} 表示的是前面的字符连续出现了5次,最多是5次最少也是5次
{n,} 代表最少出现了n次 例如 {2,} 代表最少出现2次，注意考题： {,10} --没有这个写法的

^[0-9] 表示的是以数字开头
^[a-z] 表示的是以小写字母开头
