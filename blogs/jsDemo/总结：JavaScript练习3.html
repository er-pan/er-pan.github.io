                    "width":"100px"
                },speed);

                $(".box li").stop().animate({

    <script>
        $(function(){
            var $key = 0;//这个变量用来控制播放下一张
            $("#box ol li").mousemove(function(){
                $(this).addClass('current').siblings().removeClass('current');
                // fast --快速的  slow---慢的
                $("#box ul li").eq($(this).index()).stop().fadeIn("fast").siblings().stop().fadeOut("fast");//图片跟着数字切换了
                // 当鼠标移入的时候，从新获取索引值
                $key = $(this).index();
            });

            // 开始添加定时器
            var timer = setInterval(autoplay,1500)
            // 封装autoplay函数
            function autoplay(){
                $key++;//因为一开始是第一张，我们要从第二张开始播放，所以先自加
                // 判断，要不然一直会往下走
                if($key > 5){
                    $key = 0;//此时表示第一张
                }
                // console.log($key);
                $("#box ol li").eq($key).addClass("current").siblings().removeClass('current');//有序列表自动走了
                $("#box ul li").eq($key).stop().fadeIn('fast').siblings().stop().fadeOut('fast');
            }

            // 清除定时器开始
            $("#box").hover(function(){
                // 鼠标经过清除定时器
                clearInterval(timer);
            },function(){
                // 开启定时器之前先清除定时器---设表先关
                clearInterval(timer);
                timer = setInterval(autoplay,1500);//开启定时器
            })

        })
    </script>

// .concat(数组,数组,数组...) 组合一个新的数组

// 扩展框架里面的常用方法  every筛选 和 some 类似
var arr = [5000,2000,3000];
/* 
语法格式：有三个参数
第一个参数 a -----元素的值
第二个参数 b ----- 元素的索引
第三个参数 c ----- 原数组(谁调用了这个方法那么c就是谁，此时c就是arr)
*/
var flag = arr.every(function(a,b,c){
 // console.log(arguments.length);
 // console.log(a + "=========" + b + "=======" + c);
     return a >2000;//总结：这个数组中每个元素都要大于2000的情况下，才会返回true
})
console.log(flag);//返回false 
    /* 
    大总结：every(函数) ---返回的是布尔类型，函数中有三个参数，第一个参数是元素的值，第二个参数是索引值，第三个参数原来的数组，（如果没有用到的可以省略）--特点：可以用来筛选，如果这个数组中每个元素都符合条件，最后才会返回true，如果有一项不满足条件就会返回false
    
    some() 找true，找到false会继续查找，找true，会停止查找，返回true，找不到就返回false
    
    */
// 通过every来进行筛选
var array = ["小明lklklk","小丽789456","小花abcdefg"];
var result = array.every(function(ele){
    // 数组中每个元素的长度都大于5
    return ele.length > 5
});
